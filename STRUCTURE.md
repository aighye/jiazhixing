# 目录结构

- bootstrap.php       # 应用引导文件
- index.php           # 首页入口
- login.php           # 登录页面
- logout.php          # 退出登录
- dashboard.php       # 仪表盘
- README.md           # 项目说明
- INSTALL.md          # 安装指南
- PROJECT.md          # 项目概述
-
- config/
-   └── app.php       # 应用配置文件
-
- core/               # 核心类库
-   ├── Database.php  # 数据库操作类
-   ├── Auth.php      # 用户认证类
-   ├── Logger.php    # 日志记录类
-   ├── Helper.php    # 辅助函数类
-   └── Config.php    # 配置管理类
-
- database/
-   └── install.sql   # 数据库安装脚本
-
- customers/          # 客户管理模块
-   ├── index.php     # 客户列表
-   ├── add.php       # 新增客户
-   ├── edit.php      # 编辑客户
-   ├── view.php      # 客户详情
-   └── vehicles/     # 车辆管理
-       ├── add.php
-       └── edit.php
-
- workorders/         # 维修工单模块
-   ├── index.php     # 工单列表
-   ├── add.php       # 新建工单
-   ├── view.php      # 工单详情（含派工、质检）
-   └── edit.php      # 编辑工单
-
- parts/              # 配件管理模块
-   ├── index.php     # 配件列表
-   ├── add.php       # 新增配件
-   ├── view.php      # 配件详情
-   ├── edit.php      # 编辑配件
-   └── inbound.php   # 配件入库
-
- settlements/        # 财务结算模块
-   ├── index.php     # 结算记录
-   ├── add.php       # 工单结算
-   └── view.php      # 结算详情
-
- reports/            # 统计报表模块
-   └── index.php     # 报表首页
-
- system/             # 系统管理模块
-   ├── users.php     # 用户管理
-   ├── users_save.php
-   ├── roles.php     # 角色管理
-   ├── logs.php      # 操作日志
-   └── config.php    # 系统配置
